 31 RFGraph.pluckstack = function(stk) {
 32     for (var i = 0; i < stk.length; ++i) {
 33         if (stk[i].diff == "inboth") {
 34             return stk.splice(i,1)[0];
 35         }
 36     }
 37     for (var i = 0; i < stk.length; ++i) {
 38         if (stk[i].diff == "infrom") {
 39             return stk.splice(i,1)[0];
 40         }
 41     }
 42     return stk.splice(0,1)[0];
 43 }

 47 /**
 48  * Internal computation for graph difference between two graphs
 49  * @param grf the "from" graph, as returned by makegraph
 50  * @param grt the "to" graph, as returned by makegraph
 51  */
 52 RFGraph.walkDiff = function(/**Graph*/ grf, /**Graph*/ grt) {
 53     var globalbranch = 1;
 54     var grres = {
 55         parsed: [],
 56         index: {}
 57     };
 58     
 59     bfsnumber(grf);
 60     fromnamedx = name2nodedx(grf);
 61     bfsnumber(grt);
 62     tonamedx = name2nodedx(grt);
 63     
 64     var stk = [{
 65         diff: "inboth",
 66         from: grf.topnode,
 67         to: grt.topnode
 68         }];
 69     while (stk.length > 0) {
 70         //todo: select an "inboth" or "into" path
 71         var tp = pluckstack(stk);//stk.pop();
 72         info("PLUCK", tp.to.name, tp.to.data.bfsn, tp.diff);
 73         
 74         //mark as explored
 75         tp.to.data.used = true;
 76         
 77         if (tp.diff == "inboth") {
 78             //duplicated identical node
 79             var duped = dupnode(tp.to);
 80             duped.data.diff = "inboth";
 81             duped.data.startx = tp.from.data.startx;
 82             duped.data.starty = tp.from.data.starty;
 83             
 84             grres.parsed.push(duped);
 85             grres.index[duped.id] = duped;
 86             var fromnbr = neighbors(tp.from, grf.index);
 87             var tonbr = neighbors(tp.to, grt.index);
 88             
 89             //look through every edge in the compareto graph
 90             for (var i = 0; i < tonbr.length; ++i) {
 91                 var tnbr = tonbr[i];
 92                 var dx = nodeinarr(tnbr, fromnbr);
 93                 //always connect up the "to" node
 94                 var edge = dupedge(tp.to.adjacencies[i]);
 95                 
 96                 /*
 97                 debug(tp.to.adjacencies[i].data.weight);
 98                 debug(
 99                     remnl(grt.index[tp.to.adjacencies[i].nodeFrom].name), 
100                     remnl(grt.index[tp.to.adjacencies[i].nodeTo].name)
101                 );
102                 */
103                 
104                 //if we'd like to pull edge weights from the precursor
105                 for (var j = 0; j < tp.from.adjacencies.length; ++j) {
106                     var fj = tp.from.adjacencies[j];
107                     if (grf.index[fj.nodeFrom].name == grt.index[tp.to.adjacencies[i].nodeFrom].name &&
108                         grf.index[fj.nodeTo].name == grt.index[tp.to.adjacencies[i].nodeTo].name) {
109                         //pull weight from precursor
110                         edge.data.weight = fj.data.weight;
111                     }
112                 }
113                 
114                 edge.data.diff = "into";
115                 duped.adjacencies.push(edge);
116                 if (dx > -1) {
117                     //the matching node in the from graph
118                     var fromsame = fromnbr.splice(dx, 1)[0];
119                     
120                     if (!fromsame.data.used) { fromsame.data.used = 0; }
121                     fromsame.data.used += 1;
122                     
123                     stk.push({
124                         diff: "inboth",
125                         from: fromsame,
126                         to: tnbr
127                     });
128                 } else {
129                     //XXX: this will not always work! (consider subtrees with branches)
130                     globalbranch += 1;
131                     //here, we have branched: tp.to into tnbr 
132                     //  with no matching in the "from" graph
133                     info("UNMATCHED", tp.to.data.bfsn, tp.from.name, tnbr);
134                     stk.push({
135                         diff: "into",
136                         from: tp.from, //the node we branched at in the "from" graph
137                         to: tnbr, //the new branch
138                         branchno: globalbranch
139                     });
140                 }
141             }
142             
143             //nodes remaining here correspond to deletions from the "from" graph
144             for (var i = 0; i < fromnbr.length; ++i) {
145                 var fnb = fromnbr[i];
146                 info("FROMNODE", tp.to, tp.from, fnb);
147                 
148                 //make a cross-edge from "to" graph to "from" graph
149                 duped.adjacencies.push({
150                     data: {
151                         weight: 10, //FIXME: get weight off "from" graph
152                         diff: "infrom"
153                     },
154                     nodeFrom: tp.to.id,
155                     nodeTo: fnb.id
156                 });
157                 //place the node to the "left" (arbitrarily) of the other.
158                 //fnb.data.startx = tp.to.data.startx - 400;
159                 
160                 stk.push({
161                     diff: "infrom",
162                     from: fnb,
163                     to: tp.to
164                 });
165             }
166         } else if (tp.diff == "into") {
167             var duped = dupnode(tp.to);
168             duped.data.diff = "into";
169             //XXX: see localbranch above
170             duped.data.startx = tp.from.data.startx + branchoffset[tp.branchno];
171             grres.parsed.push(duped);
172             grres.index[duped.id] = duped;
173             var tonbr = neighbors(tp.to, grt.index);
174             
175             //possibly connect up "to" neighbors
176             for (var i = 0; i < tonbr.length; ++i) {
177                 var tnbr = tonbr[i];
178                 
179                 var candidates = fromnamedx[tnbr.name];
180                 var tomerge = null;
181                 if (candidates) {
182                     for (var j = 0; j < candidates.length; ++j) {
183                         var cand = candidates[j];
184                         //heuristic for linking.
185                         var okay = !cand.data.used || cand.data.used < cand.data.inlinks;
186                         
187                         //XXX: Magic Tuning Parameter (BFS lookahead)!
188                         if (cand.data.bfsn - tp.from.data.bfsn < BFSLOOKAHEAD && okay) {
189                             info("MERGE ", cand.name, cand.data.bfsn, tp.from.data.bfsn);
190                             tomerge = cand;
191                             
192                             if (!cand.data.used) { cand.data.used = 0; }
193                             cand.data.used += 1;
194                         } 
195                     }
196                 }
197                 
198                 if (!tonbr[i].data.used) {
199                     duped.adjacencies.push(dupedge(tp.to.adjacencies[i]));
200                     
201                     if (null == tomerge) {
202                         stk.push({
203                             diff: "into",
204                             from: tp.from,
205                             to: tnbr,
206                             branchno: tp.branchno
207                         });
208                     } else {
209                         globalbranch -= 1;
210                         info("MERGEOK", tomerge, tnbr);
211                         stk.push({
212                             diff: "inboth",
213                             from: tomerge,
214                             to: tnbr
215                         });
216                     }
217                 }
218             }
219         } else if (tp.diff == "infrom") {
220             var duped = dupnode(tp.from);
221             duped.data.diff = "infrom";
222             grres.parsed.push(duped);
223             grres.index[duped.id] = duped;
224             
225             var fromnbr = neighbors(tp.from, grf.index);
226             
227             for (var i = 0; i < fromnbr.length; ++i) {
228                 var fnbr = fromnbr[i];
229                 var candidates = tonamedx[fnbr.name];
230                 var tomerge = null;
231                 
232                 if (candidates) {
233                     for (var j = 0; j < candidates.length; ++j) {
234                         var cand = candidates[j];
235                         info(cand.data.bfsn, tp.to.data.bfsn, cand.name, tp.from.name);
236                         var bdiff = cand.data.bfsn - tp.to.data.bfsn;
237                         if (bdiff < BFSLOOKAHEAD) {
238                             info("FROMMERGE ", cand.name, cand.data.bfsn, tp.from.data.bfsn);
239                             tomerge = cand;
240                             break;
241                         } 
242                     }
243                 }
244                 
245                 if (null == tomerge) {
246                     var edge = dupedge(tp.from.adjacencies[i]);
247                     edge.data.diff = "infrom";
248                     duped.adjacencies.push(edge);
249                     stk.push({
250                         diff: "infrom",
251                         from: fnbr,
252                         to: tp.to
253                     });
254                 } else {
255                     info("MERGEOK", fnbr.name, tomerge.name, tomerge.data.bfsn);
256                     globalbranch -= 1;
257                     //make a cross-edge from "from" graph to "to" graph
258                     /*
259                     duped.adjacencies.push({
260                         data: {
261                             weight: 10, //FIXME: get weight off "from" graph
262                             diff: "infrom"
263                         },
264                         nodeFrom: tp.from.id,
265                         nodeTo: tomerge.id
266                     });*/
267                 }
268             }
269         } //end if
270     } // end while
271     
272     // first pushed item on the stack is the top node
273     grres.topnode = grres.parsed[0];
274     return grres;
275 }

285 var makeHistogram = function() {
286     //bin width for "histogram"
287     var BW = 5;
288     var dh = disthist(gres.parsed, BW);
289     var pxmap = histtopx(dh);
290     var histmax = Math.max.apply(Math, pxmap);
291     var img = mkhistpic(dh, BW, 150/histmax);
292     $("#hist-hist").empty();
293     img[0].appendTo($("#hist-hist"));
294     $("#hist-cont").css({
295         "top": img[1] + $("#info-cont").height() + 10,
296         "left": 30
297     });
298     var slide = $("#hist-slider");
299     slide.css({
300         "position": "absolute",
301         "left": -20,
302         "bottom": 0,
303         "height": img[1]
304     });
305     slide.slider({
306        "orientation": "vertical",
307        "range": "true",
308        "values": [0, img[1]],
309        "min": 0,
310        "max": img[1],
311        "slide": function(evt, ui) {
312             //look up the maximum and minimum
313             // edge lengths in the map from px -> hist values
314             MINEDGESEL = pxmap[ui.values[0]+1] * BW;
315             MAXEDGESEL = (pxmap[ui.values[1]-1]+1) * BW;
316             fd.plot();
317         }
318     });
319     MINEDGESEL = 0;
320     MAXEDGESEL = 1e10;
321 };

323 var mkhistpic = function(/**int[]*/ hist, /**int*/ bw, /**float*/ xscale) {
324     var cont = $("<div></div>").css({
325         "position": "absolute"
326     });
327     var px = 0;
328     for (var i = 0; i < hist.length; ++i) {
329         if (hist[i]) {
330             cont.append(
331               $("<div></div>").css({
332                 "position": "absolute",
333                 "left": 0,
334                 "bottom": px,
335                 "width": (i+1)*bw*xscale,
336                 "height": hist[i],
337                 "background-color": "#c00"
338               })
339             );
340             px += hist[i];
341         }
342     }
343     
344     return [cont, px];
345 }
