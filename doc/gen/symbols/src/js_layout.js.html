 31 /**
 32 @namespace Sugiyama layout
 33 Based on implementation by C. Schanck (chris at schanck dot net) of Sugiyama layout algorithm
 34 */
 35 var Layout = function(/**Graph*/ g, /**string*/ name) {
 36     this.name = name;
 37     this.g = g;
 38 };
 39 
 40 Layout.prototype.compute = function() {
 41     var g = this.g;
 42     var levels = this.levelFillWrap();
 43     this.levels = levels;
 44 
 45     this.gridsize = 0;
 46     for (var i = 0; i < this.levels.length; ++i) {
 47         this.gridsize = Math.max(this.gridsize, levels[i].length);
 48     }
 49     this.info("gridsize=", this.gridsize);
 50     
 51     //solve edge crosses
 52     this.uncrossAll(g, levels);
 53     
 54     //move to barycenter
 55     this.moveNodes(g, levels, this.gridsize);
 56     
 57     //move nodes into place based on grid position
 58     this.assignPos(levels, 400, 60);
 59 
 60     this.applyExtended(g);
 61 }
 62 
 63 Layout.prototype.applyExtended = function(g) {
 64     var thisex = $.extend({}, LAYOUTEX["all"], LAYOUTEX[this.name]);
 65     for (var i = 0; i < g.parsed.length; ++i) {
 66         var n = g.parsed[i];
 67         if (n.id in thisex) {
 68             n.data.startx = thisex[n.id].x;
 69         }
 70     }
 71 }

 73 var maxUndef = function() {
 74     var args = arguments;
 75     for (var i = 0; i < args.length; ++i) {
 76         if (!args[i]) {
 77             args[i] = 0;
 78         }
 79     }
 80     return Math.max.apply(Math, args);
 81 }
 82 
 83 var minUndef = function() {
 84     var args = arguments;
 85     for (var i = 0; i < args.length; ++i) {
 86         if (!args[i]) {
 87             args[i] = 0;
 88         }
 89     }
 90     return Math.min.apply(Math, args);
 91 }

 93 Layout.FROM = 1;
 94 Layout.TO = 2;
 95 Layout.BOTH = 4;

 97 Layout.lr = PAPER ? 100 : 200;
 98 Layout.basedist = 10;
 99 Layout.efn = function(l, basedist) {
100     //return l/200 + basedist * 2;
101     //return l/500 + basedist;
102     return (2.0/(1.0+Math.exp(-l/Layout.lr))-1.0) * Layout.lr * 2 + ((PAPER) ? 35 : Layout.basedist);
103 };

105 /**
106  * @param basedist the shortest node-node distance
107  * @param computeflags bit string with flags from Layout.FROM, Layout.TO, Layout.BOTH
108  */
109 Layout.prototype.longestPath = function(/**Number*/ basedist, /**Array*/ levels, /**int*/ computeflags) {
110     var g = this.g, n = g.topnode;
111     //flatten the topological sort
112     var topo = [];
113     for (var i = 0; i < levels.length; ++i) {
114         for (var j = 0; j < levels[i].length; ++j) {
115             var n = levels[i][j];
116             //length to in combined graph
117             if (computeflags & Layout.BOTH) {
118                 n.data.lentob = 0;
119                 n.data.critpb = 0;
120             }
121             //length to in "to" graph
122             if (computeflags & Layout.TO) { //&&!(n.data.diff == "into")
123                 n.data.lentot = 0;
124                 n.data.critpt = 0;
125             }
126             //length to in "from" graph
127             if (computeflags & Layout.FROM) {
128                 n.data.lentof = 0;
129                 n.data.critpf = 0;
130             }
131 
132             topo.push(n);
133         }
134     }
135 
136     Layout.basedist = basedist;
137 
138     //TODO: keep a queue of unconverged nodes
139     for (var loop = 0; loop < 10; ++loop) {
140         for (var i = 0; i < topo.length; ++i) {
141             var v = topo[i];
142             for (var j = 0; j < v.adjacencies.length; ++j) {
143                 var e = v.adjacencies[j];
144                 var w = g.index[(e.nodeFrom == v.id) ? e.nodeTo : e.nodeFrom];
145 
146                 if (w) {
147                     var wboth = maxUndef(e.data.weightf, e.data.weightt);
148                     if (computeflags & Layout.BOTH) {
149                         if (!e.data.scwboth) {
150                             e.data.scwboth = Layout.efn(wboth);
151                         }
152                         if (w.data.lentob <= v.data.lentob + e.data.scwboth) {
153                             w.data.lentob = v.data.lentob + e.data.scwboth;
154                         }
155                         if (w.data.critpb <= v.data.critpb + wboth) {
156                             w.data.critpb = v.data.critpb + wboth;
157                         }
158                     }
159                     if (computeflags & Layout.TO) {
160                         if (e.data.weightt !== undefined) {
161                             if (e.data.scweightt === undefined) {
162                                 e.data.scweightt = Layout.efn(e.data.weightt);
163                             }
164                             if (w.data.lentot <= v.data.lentot + e.data.scweightt) {
165                                 w.data.lentot = v.data.lentot + e.data.scweightt;
166                             }
167                             if (w.data.critpt <= v.data.critpt + e.data.weightt) {
168                                 w.data.critpt = v.data.critpt + e.data.weightt;
169                             }
170                         } else {
171                             this.info("No to weight for edge when computing Layout.TO ", e);
172                         }
173                     }
174                     if (computeflags & Layout.FROM) {
175                         if (e.data.weightf !== undefined) {
176                             if (e.data.scweightf === undefined) {
177                                 e.data.scweightf = Layout.efn(e.data.weightf);
178                             }
179                             if (w.data.lentof <= v.data.lentof + e.data.scweightf) {
180                                 w.data.lentof = v.data.lentof + e.data.scweightf;
181                             }
182                             if (w.data.critpf <= v.data.critpf + e.data.weightf) {
183                                 w.data.critpf = v.data.critpf + e.data.weightf;
184                             }
185                         } else {
186                             this.info("No from weight for edge when computing Layout.FROM", e);
187                         }
188                     }
189                 } else {
190                     this.info("Warning: no node found in graph for " + e.nodeTo);
191                 }
192             }
193         }
194     }
195 }

207 /**
208 Non-recursive version of levelFill
209 */
210 Layout.prototype.levelFillStack = function(/**Graph*/ g, /**Node*/ n, /**Array[Array]*/ levels) {
211     var stk = [];
212     stk.push({lvl: 0, n: n});
213     
214     while (stk.length > 0) {
215         var top = stk.pop();
216         var n = top.n;
217         var curlvl = top.lvl;
218         
219         if (!n.data[this.name]) {
220             if (!levels[curlvl]) {
221                 levels[curlvl] = [];
222             }
223             levels[curlvl].push(n);
224             n.data[this.name] = {
225                 levelno: curlvl,
226                 crossno: levels[curlvl].length,
227                 crossadd: 0,
228                 priority: 0,
229                 gridpos: 0
230             };
231             
232             //for (var i = 0; i < n.adjacencies.length; ++i) {
233             for (var i = n.adjacencies.length-1; i >= 0; --i) {
234                 var adj = n.adjacencies[i];
235                 var other = g.index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom];
236                 stk.push({lvl: curlvl+1, n: other});
237             }
238         }
239     }
240     
241     return levels;
242 }

244 /**
245 Non-recursive version of levelFill
246 */
247 Layout.prototype.levelFillStackDAG = function(/**Graph*/ g, /**Node*/ n, /**Array[Array]*/ levels) {
248     var stk = [];
249     stk.push({lvl: 0, n: n});
250     
251     for (var i = 0; i < g.parsed.length; ++i) {
252         var n = g.parsed[i];
253         n.data[this.name] = {
254             levelno: 0,
255             crossno: 0, //levels[curlvl].length,
256             crossadd: 0,
257             priority: 0,
258             gridpos: 0
259         };       
260     }
261 
262     var iter = 0;
263     while (stk.length > 0) {
264         ++iter;
265         var top = stk.pop();
266         var n = top.n;
267         var curlvl = top.lvl;
268         n.data[this.name].levelno = Math.max(curlvl, n.data[this.name].levelno);
269         
270         //for (var i = 0; i < n.adjacencies.length; ++i) {
271         for (var i = n.adjacencies.length-1; i >= 0; --i) {
272             var adj = n.adjacencies[i];
273             var other = g.index[adj.nodeTo]; //g.index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom];
274             stk.push({lvl: curlvl+1, n: other});
275         }
276     }
277 
278     var lvldx = 0;
279     for (var i = 0; i < g.parsed.length; ++i) {
280         var n = g.parsed[i];
281         var curlvl = n.data[this.name].levelno;
282         while (lvldx <= curlvl) {
283             if (!levels[lvldx]) {
284                 levels[lvldx] = [];
285             }
286             ++lvldx
287         }
288         levels[curlvl].push(n);
289         n.data[this.name].crossno = levels[curlvl].length;
290     }
291     
292     return levels;
293 }

295 /**
296 Recursive version of levelFill
297 */
298 Layout.prototype.levelFill = function(/**Graph*/ g, /**Node*/ n, /**Array[Array]*/ levels, /**int*/ curlvl) {
299     if (levels.length == curlvl) {
300         levels.push([]);
301     }
302     
303     if (!n.data[this.name]) {
304         levels[curlvl].push(n);
305         n.data[this.name] = {
306             levelno: curlvl,
307             crossno: levels[curlvl].length,
308             crossadd: 0,
309             priority: 0,
310             gridpos: 0
311         };
312        
313         for (var i = 0; i < n.adjacencies.length; ++i) {
314             var adj = n.adjacencies[i];
315             var other = g.index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom];
316             this.levelFill(g, other, levels, curlvl + 1);
317         }
318     }
319     
320     return levels;
321 }

323 Layout.prototype.uncrossAll = function(/**Graph*/ g, /**Array[Array]*/ levels) {
324     for (var iterations = 0; iterations < Layout.MAX_ITERATIONS; ++iterations) {
325         var movements = 0;
326         
327         for (var i = 0; i < levels.length-1; ++i) {
328             movements += this.uncrossLayer(g, levels[i], i, true);
329         }
330         
331         for (var i = levels.length-1; i > 0; --i) {
332             movements += this.uncrossLayer(g, levels[i], i, false);
333         }
334         
335         this.info("uncross movements:", movements);
336         if (0 == movements) {
337             break;
338         }
339     }
340 }

342 Layout.prototype.crossind = function(/**Node*/ n) {
343     return (n.data[this.name].crossadd == 0) ? 0 : n.data[this.name].crossno / n.data[this.name].crossadd;
344 }

346 Layout.prototype.uncrossLayer = function(/**Graph*/ g, /**Array*/ level, /**int*/ leveldx, /**bool*/ down) {
347     var self = this;
348     var movements = 0;
349     
350     var prevsort = level.slice(0);
351     level.sort(function(a, b) {
352         return self.crossind(b) - self.crossind(a);
353     });
354     
355     for (var i = 0; i < level.length; ++i) {
356         if (level[i].data[this.name].crossno != prevsort[i].data[this.name].crossno) {
357             ++movements;
358         }
359     }
360     
361     this.info("uncrosslayer movements=", movements);
362     
363     for (var i = level.length-1; i > -1; --i) {
364         var n = level[i];
365         
366         var iterarr = down ? n.adjacencies : n.incoming;
367         for (var j = 0; j < iterarr.length; ++j) {
368             var adj = iterarr[j];
369             var other = g.index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom];
370             
371             //this condition should always fail... 
372             if (!other) {
373                 continue;
374             }
375             
376             if ( (down && other.levelno > leveldx) || (!down && other.levelno < leveldx) ) {
377                 other.data[this.name].crossno += this.crossind(n);
378                 other.data[this.name].crossadd += 1;
379             }
380         }
381     }
382     
383     return movements;
384 }

386 Layout.prototype.moveNodes = function(/**Graph*/ g, /**Array[Array]*/ levels, /**int*/ gridsize) {
387     for (var i = 0; i < g.parsed.length; ++i) {
388         var n = g.parsed[i];
389         
390         for (var j = 0; j < n.adjacencies.length + n.incoming.length; ++j) {
391             var adj = (j < n.adjacencies.length) ? n.adjacencies[j] : n.incoming[j-n.adjacencies.length];
392             var other = g.index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom];
393             
394             //this condition should always fail... 
395             if (!other) {
396                 continue;
397             }
398             
399             if (other.levelno != n.levelno) {
400                 n.data[this.name].priority += 1;
401             }
402         }
403     }
404     for (var i = 0; i < levels.length; ++i) {
405         for (var j = 0; j < levels[i].length; ++j) {
406             levels[i][j].data[this.name].gridpos = j;
407         }
408     }
409     
410     for (var iterations = 0; iterations < Layout.MAX_ITERATIONS; ++iterations) {
411         var movements = 0;
412         
413         for (var i = 0; i < levels.length-1; ++i) {
414             movements += this.moveLevel(g, levels[i], i, gridsize);
415         }
416         
417         for (var i = levels.length-1; i > 0; --i) {
418             movements += this.moveLevel(g, levels[i], i, gridsize);
419         }
420         
421         this.info("movenodes movements:", movements);
422         if (0 == movements) {
423             break;
424         }
425     }
426 }

428 Layout.prototype.moveLevel = function(/**Graph*/ g, /**Array*/ level, /**int*/ levelno, /**int*/ gridsize) {
429     var movements = 0;
430     
431     for (var i = 0; i < level.length; ++i) {
432         var n = level[i];
433         var possum = 0;
434         var nnodes = 0;
435         
436         for (var j = 0; j < n.adjacencies.length + n.incoming.length; ++j) {
437             var adj = (j < n.adjacencies.length) ? n.adjacencies[j] : n.incoming[j-n.adjacencies.length];
438             var other = g.index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom];
439             
440             possum += n.data[this.name].gridpos;
441             ++nnodes;
442         }
443         
444         if (nnodes > 0) {
445             var newpos = Math.round(possum/nnodes);
446             var toright = newpos > n.data[this.name].gridpos;
447             var moved = true;
448             while (newpos != n.data[this.name].gridpos && moved) {
449                 moved = this.move(toright, level, i, n.data[this.name].priority, gridsize);
450                 movements += (moved) ? 1 : 0;
451             }
452         }
453     }
454     
455     return movements;
456 }

458 Layout.prototype.move = function(/**bool*/ toright, /**Array*/ level, /**int*/ dx, /**int*/ priority, /**int*/ gridsize) {
459     var n = level[dx];
460     var moved = false;
461     var targetdx = dx + (toright) ? 1 : -1;
462     var targetgp = n.data[this.name].gridpos + (toright) ? 1 : -1;
463     
464     if (targetgp < 0 || targetgp >= gridsize) {
465         return false;
466     }
467     
468     if ( (toright && dx == level.length - 1) || (!toright && dx == 0) ) {
469         moved = true;
470     } else {
471         var n2 = level[targetdx];
472         if (n2.data[this.name].gridpos == targetgp) {
473             if (n2.data[this.name].priority >= priority) {
474                 return false;
475             } else {
476                 moved = this.move(toright, level, targetdx, priority, gridsize);
477             }
478         } else {
479             moved = true;
480         }
481     }
482     
483     if (moved) {
484         n.data[this.name].gridpos = targetgp;
485     }
486     return moved;
487 }

489 Layout.prototype.assignPos = function(/**Array[Array]*/ levels, /**int*/ hspace, /**int*/ vspace) {
490     for (var i = 0; i < levels.length; ++i) {
491         for (var j = 0; j < levels[i].length; ++j) {
492             var n = levels[i][j];
493             
494             n.data.startx = n.data[this.name].gridpos * hspace;
495             n.data.starty = n.data[this.name].levelno * vspace;
496         }
497     }
