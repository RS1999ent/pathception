 43 /**
 44 Override the default jit label to change the rendered label
 45 (e.g. add additional information for debugging/viewing purposes)
 46 */
 47 JitExtend.renderLabel = function(canvas, node, controller) {
 48     var ctx = canvas.getCtx();
 49     ctx.font = FONTSIZE + "px Arial bold"; // + " Segoe UI";
 50     if (PAPER) {
 51         ctx.font = FONTSIZE + "px Myriad Pro Black";
 52     }
 53     var pos = node.pos.getc(true);
 54     
 55     //var name = node.name + " " + node.data.bfsn;
 56     var name = node.name;
 57     
 58     ctx.fillStyle = "rgba(0,0,0," + JitExtend.getalpha(node.data.diff) + ")";
 59     ctx.fillText(name, pos.x, pos.y + node.getData("height") / 2 + (PAPER ? 3 : 0));
 60 };
 61 $jit.ForceDirected.Label.Native.prototype.renderLabel = JitExtend.renderLabel;

 63 /**
 64 Additional Edge types:
 65 <dl>
 66 <dt>label-line</dt>
 67     <dd>Render text on the edge according to edge.data.weight. Also render graph diff colors</dd>
 68 </dl>
 69 */
 70 JitExtend.addlEdgeTypes = {  
 71   'label-line': {  
 72     'render': function(adj, canvas) {  
 73         var no1 = adj.nodeFrom, p1 = no1.pos.getc(true);
 74         var no2 = adj.nodeTo, p2 = no2.pos.getc(true);
 75         var ctx = canvas.getCtx();
 76 
 77         var diff = adj.data.diff;
 78         var weightf = adj.data.weightf;
 79         var weightt = adj.data.weightt;
 80 
 81         if (diff) {
 82             ctx.strokeStyle = ctx.fillStyle = "rgba(200,200,200," + JitExtend.getalpha(diff) + ")";
 83             if (SHOWBOTH) {
 84                 if (diff == "infrom") {
 85                     ctx.strokeStyle = JitExtend.PRECOL;
 86                 } else if (diff == "into") {
 87                     ctx.strokeStyle = JitExtend.MUTCOL;
 88                 }
 89             }
 90         }
 91         
 92         if (SEL_ON) {
 93             if (p1.x > SEL_REL_LEFT && p1.x < SEL_REL_LEFT + SEL_REL_W &&
 94                 p1.y > SEL_REL_TOP && p1.y < SEL_REL_TOP + SEL_REL_H &&
 95                 p2.x > SEL_REL_LEFT && p2.x < SEL_REL_LEFT + SEL_REL_W &&
 96                 p2.y > SEL_REL_TOP && p2.y < SEL_REL_TOP + SEL_REL_H) {
 97                 ctx.strokeStyle = "#aa0000";
 98             }
 99         }
100 
101         var fliporder = false;
102         var isparl = (diff == "parl"); // is a "parallel lines" line
103         //hide parallel lines edges from other views
104         if (isparl && !SHOWSBS) {
105             return;
106         }
107 
108         var w = maxUndef(weightf, weightt);
109         if (w) {
110             if (w < MINEDGESEL || w > MAXEDGESEL) {
111                 return;
112             }
113 
114             //don't show edge labels
115             if (false) {
116                 //show both the "from" and "to" weights
117                 ctx.save();
118                 var px = (p1.x + p2.x)/2 + 12;
119                 var py = (p1.y + p2.y)/2;
120                 var txt = weightf ? weightf + "" : "";
121                 txt += " / ";
122                 txt += weightt ? weightt + "" : "";
123                 txt += " us";
124 
125                 ctx.fillText(txt, px, py+FONTSIZE/2);
126                 //this.nodeHelper.circle.render('fill', dotc, 3, canvas); 
127                 ctx.restore();
128             }
129         }
130         
131         //flip node order if directionality is reversed
132         /* //from level numbering
133         if (adj.nodeFrom.data.lb && adj.nodeTo.data.lb) {
134             fliporder = (adj.nodeFrom.data.lb.levelno >= adj.nodeTo.data.lb.levelno);
135         }
136         */
137         if ((adj.nodeFrom.data.lentob !== undefined) && (adj.nodeTo.data.lentob !== undefined)) {
138             fliporder = (adj.nodeFrom.data.lentob >= adj.nodeTo.data.lentob);
139         }
140 
141         var n1 = (fliporder) ? no2 : no1,
142             q1 = (fliporder) ? p2 : p1;
143         var n2 = (fliporder) ? no1 : no2,
144             q2 = (fliporder) ? p1 : p2;
145         //thin lines for backwards edges
146         if (q2.y < q1.y) {
147             ctx.lineWidth = 10;
148         } else {
149             ctx.lineWidth = 20;
150         }
151 
152         //draw edges starting at end of boxes
153         // (these need tweaking based on node height)
154         var q1x = q1.x;
155         var q2x = q2.x;
156         if (false && isparl && SHOWSBS) {
157             q1x += JitExtend.getNodeWidth(adj.nodeFrom)/2;
158             q2x -= JitExtend.getNodeWidth(adj.nodeFrom)/2;
159         }
160         var q1y = q1.y + ((isparl) ? 0 : 15);
161         var q2y = q2.y - ((isparl) ? 0 : 5);
162 
163         //red highlight for statistically significant edges
164         if (diff && (adj.data.colort == "red")) {
165             var origlw = ctx.lineWidth;
166             var origstroke = ctx.strokeStyle;
167             ctx.strokeStyle = "rgba(255,0,0," + JitExtend.getalpha(diff) + ")";
168             ctx.lineWidth = origlw * 2;
169             ctx.beginPath();
170             ctx.moveTo(q1x,q1y);
171             ctx.lineTo(q2x,q2y);
172             ctx.stroke();
173 
174             ctx.lineWidth = origlw;
175             ctx.strokeStyle = origstroke;
176         }
177         
178         if (isparl) {
179             //parallel lines
180             ctx.beginPath();
181             ctx.moveTo(q1x, q1y);
182             ctx.lineWidth = 4;
183             ctx.strokeStyle = "rgba(170,170,170,1)";
184             var PTS = 20;
185             var space = 4;
186             if (q1x > q2x) {
187                 var tmpx = q1x, tmpy = q1y;
188                 q1x = q2x; q1y = q2y;
189                 q2x = tmpx; q2y = tmpy;
190             }
191             var deltax = (q2x-q1x-space*PTS)/PTS;
192             var deltay = (q2y-q1y)/PTS;
193             for (var x = 0; x < PTS; ++x) {
194                 q1x += space;
195                 ctx.moveTo(q1x,q1y);
196                 ctx.lineTo(q1x+=deltax,q1y+=deltay);
197             }
198             ctx.stroke();
199         } else {
200             var fracsolid = 1;
201             if (SHOWBOTH) {
202                 fracsolid = adj.data.scwboth / (n2.data.lentob - n1.data.lentob);
203             } else if (diff == "infrom" || (diff == "inboth" && SHOWSBS)) {
204                 fracsolid = adj.data.scweightf / (n2.data.lentof - n1.data.lentof);
205             } else if (diff == "into" || diff == "g2") {
206                 fracsolid = adj.data.scweightt / (n2.data.lentot - n1.data.lentot);
207             } else {
208                 fracsolid = 
209                     adj.data.scweightf / (n2.data.lentof - n1.data.lentof) * FRACFROM +
210                     adj.data.scweightt / (n2.data.lentot - n1.data.lentot) * FRACTO;
211                 fracsolid /= (FRACFROM + FRACTO);
212             }
213             if (isNaN(fracsolid)) {
214                 fracsolid = 1;
215             }
216             var FRACSOLID_THRESH = 0.99;
217             var qix = q1x + (q2x - q1x) * fracsolid;
218             var qiy = q1y + (q2y - q1y) * fracsolid;
219 
220             if (SHOWBOTH && (diff == "inboth") && 
221                 weightf !== undefined && weightt !== undefined) {
222                 //diff lines
223                 ctx.beginPath();
224                 var space = 6;
225                 ctx.moveTo(q1x-space, q1y);
226                 ctx.lineWidth = space*2;
227                 ctx.strokeStyle = JitExtend.PRECOL;
228                 var targx = (weightf < weightt) ? (qix - q1x) * (weightf / weightt) + q1x - space : qix-space;
229                 var targy = (weightf < weightt) ? (qiy - q1y) * (weightf / weightt) + q1y : qiy; 
230                 ctx.lineTo(targx,targy);
231                 ctx.stroke();
232                 if (fracsolid < FRACSOLID_THRESH) {
233                     ctx.beginPath();
234                     ctx.moveTo(targx,targy);
235                     ctx.lineWidth *= 0.3;
236                     ctx.lineTo(q2x-space, q2y);
237                     ctx.stroke();
238                 }
239 
240                 ctx.beginPath();
241                 ctx.moveTo(q1x+space, q1y);
242                 ctx.lineWidth = space*2;
243                 ctx.strokeStyle = JitExtend.MUTCOL;
244                 targx = (weightf < weightt) ? qix+space : (qix - q1x) * (weightt / weightf) + q1x + space;
245                 targy = (weightf < weightt) ? qiy : (qiy - q1y) * (weightt / weightf) + q1y;
246                 ctx.lineTo(targx,targy);
247                 ctx.stroke();
248                 if (fracsolid < FRACSOLID_THRESH) {
249                     ctx.beginPath();
250                     ctx.moveTo(targx,targy);
251                     ctx.lineWidth *= 0.3;
252                     ctx.lineTo(q2x+space, q2y);
253                     ctx.stroke();
254                 }
255             } else {
256                 //regular lines
257                 ctx.beginPath();
258                 ctx.moveTo(q1x, q1y);
259                 ctx.lineTo(qix, qiy);
260                 ctx.stroke();
261                 if (fracsolid < FRACSOLID_THRESH) {
262                     ctx.beginPath();
263                     ctx.moveTo(qix, qiy);
264                     ctx.lineWidth *= 0.3;
265                     ctx.lineTo(q2x, q2y);
266                     ctx.stroke();
267                 }
268             }
269         }
270     }
271     //,'contains': function(adj, pos) {  
272     //  
273     //}  
274   },
275   //side-by-side length comparison lines
276   'compare-line': {
277     'render': function(adj, canvas) {
278         var p1 = adj.nodeFrom.pos.getc(true);
279         var p2 = adj.nodeTo.pos.getc(true);
280         var ctx = canvas.getCtx();
281         
282         if (adj.data.wpre && adj.data.wmut) {
283             //fraction of the edge length for pre and mut components
284             var prefac = adj.data.wpre / adj.data.wmut;
285             var mutfac = 1.0;
286             if (prefac > 1.0) {
287                 mutfac = 1.0 / prefac;
288                 prefac = 1.0;
289             }
290             var dx = p2.x - p1.x;
291             var dy = p2.y - p1.y;
292             
293             var eoffset = 2;
294             var w1 = 4;
295             var w2 = 4;
296             if (adj.data.spre && adj.data.smut) {
297                 //standard deviations for pre and mut components
298                 w1 = adj.data.spre;
299                 w2 = adj.data.smut
300                 eoffset = (w1+w2)/4.0;
301             }
302             
303             ctx.lineWidth = w1;
304             this.edgeHelper.line.render(
305                 {x: p1.x-eoffset, y: p1.y},
306                 {x: p1.x+dx*prefac-eoffset, y: p1.y+dy*prefac}, canvas);
307             
308             ctx.lineWidth = w2;
309             this.edgeHelper.line.render(
310                 {x: p1.x+eoffset, y: p1.y},
311                 {x: p1.x+dx*mutfac+eoffset, y: p1.y+dy*mutfac}, canvas);
312         }
313     }
314   }
315 };
316 $jit.ForceDirected.Plot.EdgeTypes.implement(JitExtend.addlEdgeTypes);

327 JitExtend.getalpha = function(diff) {
328     if (diff) {
329         if (diff == "inboth") {
330             if (SHOWBOTH) {
331                 //return Math.max(0.4,2 - (FRACFROM + FRACTO));
332                 return 1.0;
333             } else {
334                 return Math.max(FRACFROM, FRACTO);
335             }
336         } else if (diff == "into") {
337             return FRACTO;
338         } else if (diff == "infrom") {
339             return FRACFROM;
340         } else if (diff == "g2") {
341             return SHOWSBS ? 1.0 : 0.0;
342         }
343     }
344     return 1;
345 }

352 /**
353 Additional Node types:
354 <dl>
355 <dt>rf-node</dt>
356     <dd>Request flow node: render a color/stroke based on the graph diff (node.data.diff)
357         and compute space occupancy (with contains)</dd>
358 </dl>
359 */
360 JitExtend.addlNodeTypes = {  
361   'rf-node': {  
362     'render': function(node, canvas) {
363         var p = node.pos.getc(true);
364         
365         var ctx = canvas.getCtx();
366         ctx.save();
367         var w = JitExtend.getNodeWidth(node) * (PAPER ? 1.5 : 1), h = FONTSIZE + (PAPER ? 10 : 8);
368 
369         var nc = "rgba(0,255,0,";
370         if (node.data.lblcol) {    
371             var c = node.data.lblcol;
372             nc = c.replace("rgb","rgba").substring(0,c.length) + ",";
373         } 
374         if (node.data.diff) {
375             if ((node.data.diff == "inboth") || !SHOWBOTH || (node.data.diff == "g2")) {
376                 ctx.fillStyle = ctx.strokeStyle = nc + JitExtend.getalpha(node.data.diff) + ")";
377                 ctx.fillRect(p.x - w/2, p.y - h/2 + 5, w, h, canvas);
378                 ctx.lineWidth = 2;
379             } else if (node.data.diff == "into") {
380                 ctx.fillStyle = nc + FRACTO + ")";
381                 ctx.fillRect(p.x - w/2, p.y - h/2 + 5, w, h, canvas);
382                 //ctx.strokeStyle = "rgba(255,0,0," + JitExtend.getalpha(node.data.diff) + ")";
383                 ctx.strokeStyle = JitExtend.MUTCOL;
384                 ctx.lineWidth = 5;
385                 ctx.strokeRect(p.x - w/2, p.y - h/2 + 5, w, h);
386                 ctx.drawImage(IMG_TO, p.x - w/2 - 20, p.y - h/2);
387             } else if (node.data.diff == "infrom") {
388                 ctx.fillStyle = nc + FRACFROM + ")";
389                 ctx.fillRect(p.x - w/2, p.y - h/2 + 5, w, h, canvas);
390                 ctx.strokeStyle = JitExtend.PRECOL;
391                 ctx.lineWidth = 5;
392                 ctx.strokeRect(p.x - w/2, p.y - h/2 + 5, w, h);
393                 ctx.drawImage(IMG_FROM, p.x - w/2 - 20, p.y - h/2 - 5);
394             }
395         } else {
396             ctx.fillRect(p.x - w/2, p.y - h/2 + 5, w, h, canvas);
397         }
398         
399         ctx.restore();
400     }
401     ,'contains': function(node, pos) {
402         var p = node.pos.getc(true);
403         var w = JitExtend.getNodeWidth(node), h = FONTSIZE + 8;
404         
405         return (pos.x < p.x + w/2) && (pos.x > p.x - w/2) && 
406             (pos.y < p.y + h/2) && (pos.y > p.y - h/2);
407     }  
408   }  
409 };

411 JitExtend.getNodeWidth = function(node) {
412     return node.name.length*10+FONTSIZE+8;
413 }
