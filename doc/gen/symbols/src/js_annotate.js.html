 32 /**
 33  * Annotation tool
 34  */
 35 var Annotate = function(contsel,btnsel,radiosel) {
 36 	this.contsel = contsel;
 37 	this.btnsel = btnsel;
 38 	this.radiosel = radiosel;
 39 	var self = this;
 40 	$(this.btnsel).button({
 41 		icons: {primary: "ui-icon-pin-s"}
 42 	}).click(function(e) {
 43 		self.doMark(self.gp.x, self.gp.y, self.getCheckedType());
 44 		//XXX: this should really be non-global
 45 		fd.plot();
 46 	});
 47 
 48 	this.txtbtnify("#btn-anno-represent","represent",
 49 		"Please describe in a sentence or two what these changes represent.");
 50 	this.txtbtnify("#btn-anno-interesting","interesting",
 51 		"Is there anything else about the graph that you find interesting or problematic?");
 52 
 53 	this.shown = [];
 54 	this.marks = [];
 55 }
 56 
 57 Annotate.prototype.getCheckedType = function() {
 58 	return $("input:radio[name=" + this.radiosel + "]:checked").val();
 59 };
 60 
 61 Annotate.prototype.txtbtnify = function(elsel,key,desc) {
 62 	var SMFONT = "0.9em";
 63 	var self = this;
 64 	$(elsel).button({
 65 		icons: {primary: "ui-icon-pencil"}
 66 	}).css({
 67 		"font-size": SMFONT
 68 	}).click(function(e) {
 69 		var thetxt = prompt(desc,"");
 70 		self.markStr(key, thetxt);
 71 	});
 72 };
 73 
 74 /**
 75  * Mark a string with no coordinates
 76  */
 77 Annotate.prototype.markStr = function(/**string*/ key, /**string*/ val) {
 78 	this.doMark(NaN, NaN, key, val);
 79 }
 80 
 81 /**
 82  * (interal) save the mark _mark_ to a file
 83  */
 84 Annotate.prototype.saveMark = function(/**Mark*/ mark) {
 85 	$.getJSON("save.php", mark, function(res) {
 86 		if (res.status != "success") {
 87 			alert("Couldn't save mark!");
 88 		}
 89 	});
 90 }
 91 
 92 /**
 93  * Show saved marks
 94  * @param task the name of the task (e.g. graphs/task3/out) to show marks for
 95  * @param path where to load marks from
 96  * @param next continuation after loaded, called with the location of the 
 97  */
 98 Annotate.prototype.showSaved = function(/**string*/ task, /**string*/ path, /**string*/ color, /**Function*/ next) {
 99 	var self = this;
100 	var sendd = {load:true};
101 	if (path) {
102 		sendd["path"] = path;
103 	}
104 	$.getJSON("save.php", sendd, function(res) {
105 		var scanfirstmode = true;
106 		var firstmode = "";
107 		for (var i = 0; i < res.length; ++i) {
108 			var ri = res[i];
109 			//mix in the color if it's specified.
110 			if (color) {
111 				ri.color = color;
112 			}
113 			if (ri.type !== "modeChange") {
114 				scanfirstmode = false;
115 			}
116 			if (scanfirstmode) {
117 				firstmode = ri.val;
118 			}
119 			ri.x = parseFloat(ri.x);
120 			ri.y = parseFloat(ri.y);
121 			if (ri.task === task) {
122 				self.shown.push(ri);
123 			}
124 		}
125 		next(firstmode);
126 	});
127 }

129 Annotate.prototype.doMark = function(x, y, type, val, color) {
130 	var newmark = {
131 		task: RFGraph.ROOTDIR,
132 		x: x,
133 		y: y,
134 		type: type,
135 		time: (new Date()).getTime()
136 	};
137 	if (val !== undefined) {
138 		newmark.val = val;
139 	}
140 	if (color !== undefined) {
141 		newmark.color = color;
142 	}
143 
144 	this.shown.push(newmark);
145 	this.marks.push(newmark);
146 	this.saveMark(newmark);
147 	this.hide();
148 }

172 Annotate.prototype.getDescription = function(st) {
173 	switch (st) {
174 	case "q1":
175 		return "Significant";
176 	case "q2":
177 		return "Latency";
178 	case "q3":
179 		return "Structural";
180 	case "q4":
181 		return "Most Impact";
182 	case "ok":
183 		return "Correct";
184 	case "fp":
185 		return "False Positive";
186 	case "fn":
187 		return "False Negative";
188 	}
189 }

195 Annotate.prototype.draw = function(canvas,alpha) {
196 	var ctx = canvas.getCtx();
197 	ctx.save();
198 	var FONTSIZE = 30;
199 	ctx.font = FONTSIZE + "px Arial bold"
200 	ctx.textBaseline = "bottom";
201 	var arrdim = 20;
202 
203 	for (var i = 0; i < this.shown.length; ++i) {
204 		var mark = this.shown[i];
205 		if (mark.type === "unmark") {
206 			var W = 180;
207 			if (mark.color) {
208 				ctx.strokeStyle = mark.color;
209 			} else {
210 				ctx.strokeStyle = "#ff0000";
211 			}
212 			ctx.lineWidth = 5;
213 			ctx.beginPath();
214 			ctx.moveTo(mark.x, mark.y - FONTSIZE);
215 			ctx.lineTo(mark.x + W, mark.y);
216 			ctx.stroke();
217 
218 			ctx.beginPath();
219 			ctx.moveTo(mark.x, mark.y);
220 			ctx.lineTo(mark.x + W, mark.y - FONTSIZE);
221 			ctx.stroke();
222 		} else if (!isNaN(mark.x)) {
223 			var desc = this.getDescription(mark.type);
224 			ctx.beginPath();
225 			ctx.moveTo(mark.x, mark.y);
226 			ctx.lineTo(mark.x, mark.y - arrdim);
227 			ctx.lineTo(mark.x + arrdim, mark.y);
228 			ctx.closePath();
229 			ctx.fillStyle = "rgba(255,0,0," + (alpha/2) + ")";
230 			ctx.fill();
231 			if (mark.color) {
232 				ctx.fillStyle = mark.color;
233 			} else {
234 				ctx.fillStyle = "rgba(255,0,0," + alpha + ")";
235 			}
236 			ctx.fillText(desc, mark.x+arrdim, mark.y);
237 		}
238 	}
239 	ctx.restore();
240 }
