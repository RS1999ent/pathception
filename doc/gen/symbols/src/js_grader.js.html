 31 /**
 32  * @param inputbox box for taking the name of the savefile
 33  * @param gobtn button to click for loading the savefile
 34  * @param anno graph annotator for loading
 35  */
 36 var Grader = function(inputbox,gobtn,anno,btnok,btnfp,btnfn) {
 37 	var self = this;
 38 	this.inputbox = inputbox;
 39 	this.gobtn = gobtn;
 40 	this.anno = anno;
 41 
 42 	$(this.gobtn).click(function() {
 43 		self.show(RFGraph.ROOTDIR);
 44 	});
 45 
 46 	$(btnok).button({
 47 		icons: {primary: "ui-icon-check"}
 48 	}).click(function(e) { self.doMark("ok"); });
 49 	
 50 	$(btnfp).button({
 51 		icons: {primary: "ui-icon-closethick"}
 52 	}).click(function(e) { self.doMark("fp"); });
 53 
 54 	$(btnfn).button({
 55 		icons: {primary: "ui-icon-closethick"}
 56 	}).click(function(e) { self.doMark("fn"); });
 57 }
 58 
 59 Grader.prototype.doMark = function(type) {
 60 	var color = "#0000ff";
 61 	this.anno.doMark(this.anno.gp.x, this.anno.gp.y, type, this.anno.getCheckedType(), color);
 62 	fd.plot();
 63 }
 64 
 65 /**
 66  * Show saved annotation results for the given task
 67  * @param task the task to get animation results for.
 68  */
 69 Grader.prototype.show = function(task) {
 70 	var self = this;
 71 	var savename = $(self.inputbox).attr("value");
 72 	self.anno.clearShown();
 73     self.anno.showSaved(task, savename, null, function(modeid) {
 74     	Mode.change(modeid);
 75         fd.plot();
 76     });
 77     //show the key
 78     var keyfile = "key/" + Mode.getCurrent() + ".txt";
 79     //keyfile = keyfile.replace(/-/ig,"");
 80     self.anno.showSaved(task, keyfile, "#00ff00", function(modeid) {
 81     	fd.plot();
 82     });
 83 }
