100 /**
101  * @param chancecb change callback
102  */
103 Mode.init = function(changecb) {
104 	var cont = $("#mode-sel-cont");
105 	cont.buttonset();
106 	Mode.onchange = changecb;
107 	cont.children().click(function(e) {
108 		var modeid = $(this).attr("id");
109 		if (modeid.length > 0) {
110 			Mode.change(modeid);
111 		}
112 	});
113 	cont.children().each(function(i, el) {
114 		if ($(el).attr("checked")) {
115 			$(el).trigger("click");
116 		}
117 	});
118 
119 	$.each(Mode.BLOCKS, function(k, v) {
120 		if (v.enabled) {
121 			Mode.showBlock(k);
122 		}
123 	});
124 
125 	//developer mode toggle
126 	var cont = $("#btn-devmode").button().click(function() {
127 		var enable = ($(this).text() == "Developer");
128 		$(this).button("option", "label", enable ? "No Developer" : "Developer");
129 		$.each(Mode.BLOCKS, function(k, v) {
130 			var method = enable ? "showBlock" : "hideBlock";
131 			var otherMethod = enable ? "hideBlock" : "showBlock";
132 			if ($.inArray(k, ["devmode","scroll","legend","tasksel","diagram"]) == -1) {
133 				Mode[method](k);
134 			}
135  
136 			Mode[otherMethod]("diagram");
137 		});
138 	});
139 };
140 
141 Mode.showBlock = function(name) {
142 	$("#" + name + "-cont").animate(Mode.BLOCKS[name].cssin);
143 };
144 Mode.hideBlock = function(name) {
145 	$("#" + name + "-cont").animate(Mode.BLOCKS[name].cssout);
146 };
147 Mode.isEnabled = function(name) {
148 	return Mode.BLOCKS[name].enabled;
149 };
150 Mode.setEnabled = function(name, val) {
151 	Mode.BLOCKS[name].enabled = val;
152 };
153 
154 Mode.getCurrent = function() {
155 	return Mode.previd;
156 }
157 
158 Mode.change = function(modeid) {
159 	Mode.onchange(modeid);
160 
161 	//unshow a mode
162 	switch(Mode.previd) {
163 	case "rad-mode-side":
164 		SHOWSBS = false;
165 		break;
166 	case "rad-mode-diff":
167 		SHOWBOTH = false;
168 		Mode.hideBlock("legend");
169 		break;
170 	case "rad-mode-anim":
171 		Mode.hideBlock("slider");
172 		$("#btn-blink").text("start");
173 		break;
174 	}
175 
176 	switch(modeid) {
177 	case "rad-mode-side":
178 		SHOWSBS = true;
179 		setslide(0, true);
180 		break;
181 	case "rad-mode-diff":
182 		SHOWBOTH = true;
183 		setslide(50, true);
184 		Mode.showBlock("legend");
185 		break;
186 	case "rad-mode-anim":
187 		Mode.showBlock("slider");
188 		setslide(100, true);
189 		//we need to re-icon when we show this.
190 		$("#btn-blink").button({
191         	icons: { primary: "ui-icon-play" }
192    		});
193 		break;
194 	default:
195 		Util.debug("Unknown mode:", modeid);
196 		break;
197 	}
198 
199 	Mode.previd = modeid;
200 };
