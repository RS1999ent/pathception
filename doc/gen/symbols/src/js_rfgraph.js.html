 36 /**
 37 Load the given precursor/mutation pair into the main window.
 38 @param prec the precursor name
 39 @param mut the mutation name
 40 @param cont what to run when loaded (gets passed the graphdiff result)
 41 */
 42 RFGraph.loadgraph = function(/**string*/ prec, /**string*/ mut, /**function*/ cont) {
 43     getgraph(prec, function(gpre) {
 44         getgraph(mut, function(gmut) {
 45             var result = dodiff(gpre, gmut);
 46             result.copymut = RFGraph.loadMutation(result.gpre, gmut);
 47             result.origpre = gpre;
 48             result.origmut = gmut;
 49             cont(result);
 50         });
 51     });
 52 }

 54 RFGraph.loadMutation = function(gpre, mutation) {
 55     var gmut = RFGraph.flatten(makegraph(mutation, {
 56         cols: gpre.cols,
 57         coldx: gpre.coldx,
 58         lblregex: /\d+us/ig,
 59         edgetimedx: 1,
 60         edgetimekey: "weightt",
 61         colorkey: "colort"
 62     }));
 63     var layt = new Layout(gmut,"lt");
 64     layt.compute();
 65     layt.longestPath(30, layt.levels, Layout.TO);
 66     GraphUtil.applyY(gmut, "lentot");
 67     //copy layout keys into lentob
 68     gmut.topnode.data.lentot = 0;
 69     for (var k in gmut.index) {
 70         var n = gmut.index[k];
 71         n.data.lentob = n.data.lentot;
 72         n.data.critpb = n.data.critpt;
 73     }
 74     return gmut;
 75 };

 80 /**
 81  * Modify node IDs in graph g by applying fn(str) -> str
 82  * Modifies g.
 83  */
 84 RFGraph.modifyNames = function(g, fn) {
 85     if (g.parsed) {
 86         for (var i = 0; i < g.parsed.length; ++i) {
 87             g.parsed[i].id = fn(g.parsed[i].id);
 88             for (var j = 0; j < g.parsed[i].adjacencies.length; ++j) {
 89                 var adj = g.parsed[i].adjacencies[j];
 90                 adj.nodeFrom = fn(adj.nodeFrom);
 91                 adj.nodeTo = fn(adj.nodeTo);
 92             }
 93         }
 94     }
 95     if (g.index) {
 96         var oldindex = g.index;
 97         g.index = {};
 98         for (var k in g.index) {
 99             g.index[fn(k)] = oldindex[k];
100         }
101     }
102     return g;
103 }

105 RFGraph.parallelLines = function(g1, g2orig, samemap) {
106     var index = g1.index;
107     var parsed = g1.parsed;
108     var g2  = RFGraph.modifyNames(g2orig, RFGraph.MOD_ADDCOPY);
109 
110     //find out max x pos of g1
111     var maxx = 0;
112     for (var i = 0; i < g1.parsed.length; ++i) {
113         var n = g1.parsed[i];
114         if (n.data.diff == undefined || n.data.diff == "infrom") {
115             maxx = Math.max(n.data.startx, maxx);
116         }
117     }
118 
119     //copy and translate right the second graph
120     var XLATEX = PAPER ? 800 : 700;
121     for (var i = 0; i < g2.parsed.length; ++i) {
122         var n = g2.parsed[i];
123         index[n.id] = n;
124 
125         n.data.startx += (maxx + XLATEX);
126         n.data.diff = "g2";
127         for (var j = 0; j < n.adjacencies.length; ++j) {
128             n.adjacencies[j].data.diff = "g2";
129         }
130 
131         parsed.push(n);
132     }
133 
134     //Add connector edges
135     for (var i = 0; i < samemap.lst.length; ++i) {
136         var it = samemap.lst[i];
137         var adj = {
138             nodeFrom: it.f.id,
139             nodeTo: RFGraph.MOD_ADDCOPY(it.t.id),
140             data: {
141                 diff: "parl"
142             }
143         };
144         index[it.f.id].adjacencies.push(adj);
145     }
146 
147     return {
148         "index": index,
149         "topnode": g1.topnode,
150         "parsed": parsed
151     };
152 };

169     //if false, defeat caching by sending a unique querystring
170     var CACHE = false;
171     $.ajax(RFGraph.ROOTDIR + "/" + id + ".json", {
172         data: {
173             "uniq": CACHE ? 0 : (new Date().getTime())
174         },
175         complete: function(obj, tdata) {
176             var txt = obj.responseText;
177             //XXX: because it's not in JSON, just eval it.
178             eval('var graph = eval(' + txt + ')');
179             cont(graph);
180         }
181     });

184 RFGraph.flatten = function(/**Graph*/ graph) {

186     var flat = Sed.dfsFlatten({whole: []}, graph.index, graph.topnode);
187     graph.flat = flat.whole;
188     return graph;

194 RFGraph.dodiff = function(/**GraphData*/ precursor, /**GraphData*/ mutation) {
195     //initial colors for request flow graphs
196     var icoldx = -1;
197     var initialcols = {
198         4: Const.COLORS[++icoldx],
199         255: Const.COLORS[++icoldx],
200         100: Const.COLORS[++icoldx],
201         10: Const.COLORS[++icoldx],
202         5100: Const.COLORS[++icoldx]
203     };
204  	var gpre = RFGraph.flatten(makegraph(precursor, {
205         cols: initialcols,
206         coldx: icoldx,
207         lblregex: /\d+us/ig,
208         edgetimedx: 0,
209         edgetimekey: "weightf",
210         colorkey: "colorf"
211     }));   
212  	var gmut = RFGraph.flatten(makegraph(mutation, {
213         cols: gpre.cols,
214         coldx: gpre.coldx,
215         lblregex: /\d+us/ig,
216         edgetimedx: 1,
217         edgetimekey: "weightt",
218         colorkey: "colort"
219     }));
220 
221     //for the original algorithm
222     //var gdiff = RFGraph.walkDiff(gpre, gmut);
223     
224     var layf = new Layout(gpre,"lf");
225     layf.levels = layf.levelFillWrap();
226     //layf.compute();
227     layf.longestPath(30, layf.levels, Layout.FROM);
228 
229     var layt = new Layout(gmut,"lt");
230     layt.levels = layt.levelFillWrap();
231     //layt.compute();
232     layt.longestPath(30, layt.levels, Layout.TO);
233 
234     var dist = Sed.editDistance(gpre, gmut);
235     var samemap = Sed.unrollSame(dist.same);
236     var gdiff = RFGraph.mergedGraph(gpre, gmut, dist.diff, samemap);
237     
238     //Layout.levels has a topological sort of the vertices
239     var layb = new Layout(gdiff,"lb");
240     layb.compute();
241     layb.longestPath(30, layb.levels, Layout.BOTH);
242     //var layblvls = layb.levels;
243     //layb.levels = layt.levels;
244     //layb.longestPath(20);
245     //layb.levels = layblvls;
246 
247     //calculate y positions (only do this once!)
248     //simpleypos(gdiff.index, gdiff.topnode);
249     GraphUtil.applyY(gdiff, "lentob");
250     //this doesn't get computed by algo
251     gdiff.topnode.data.lentot = 0;
252 
253     return { 
254         gpre: gpre,
255         gmut: gmut,
256         samemap: samemap,
257         gres: gdiff
258     };
259 }

290 /**
291  * Compute the merged graph. Modifies nodes in both the source and destination graph.
292  * Assumes root nodes are the same.
293  * @param grf source graph
294  * @param grt destination graph
295  * @param diff list of diffs from grf->grt
296  * @param samemap bidi map of same node IDs to nodes between grf and grt, as generated by Sed.unrollSame
297  */
298 RFGraph.mergedGraph = function(/**Graph*/ grf, /**Graph*/ grt, /**Array*/ diff, /**Object*/ samemap) {
299     var labelpris = {"inboth": 1, "infrom": 2, "into": 2};
300     var setlabel = function(adj, label) {
301         if ( (!labelpris[adj.data.diff]) || (labelpris[adj.data.diff] < labelpris[label]) ) {
302             adj.data.diff = label;
303         }
304     }
305     
306     //always make labels more conservative
307     var labelEdges = function(n, label) {
308         for (var j = 0; j < n.adjacencies.length; ++j) {
309             //n.adjacencies[j].data.diff = label;
310             setlabel(n.adjacencies[j],label);
311         }
312         for (var j = 0; j < n.incoming.length; ++j) {
313             //n.incoming[j].data.diff = label;
314             setlabel(n.incoming[j],label);
315         }
316     };
317     //find correspondence for edge weights
318     var diffEdges = function(nf, nflist, nt, ntlist) {
319         //this could be O(n log n) instead of O(n^2), but n is small for request flow graphs
320         for (var j = 0; j < nflist.length; ++j) {
321             var adjf = nflist[j];
322             var onf = grf.index[(adjf.nodeFrom == nf.id) ? adjf.nodeTo : adjf.nodeFrom].name;
323             var match = false;
324             for (var k = 0; k < ntlist.length; ++k) {
325                 var adjt = ntlist[k];
326                 var ont = grt.index[(adjt.nodeFrom == nt.id) ? adjt.nodeTo : adjt.nodeFrom].name;
327                 if (onf == ont) {
328                     match = adjt;
329                 }
330             }
331             if (match) {
332                 //copy over identical keys
333                 if (match.data.weightt !== undefined) {
334                     adjf.data.weightt = match.data.weightt;
335                 }
336                 if (match.data.scweightt !== undefined) {
337                     adjf.data.scweightt = match.data.scweightt;
338                 }
339                 if (match.data.colort !== undefined) {
340                     adjf.data.colort = match.data.colort;
341                 }
342             } else {
343                 adjf.data.diff = "infrom";
344             }
345         }
346     };
347     
348     //tag all inboth
349     for (var i = 0; i < samemap.lst.length; ++i) {
350         samemap.lst[i].f.data.diff = "inboth";
351         //Even if all the edges are not inboth, they will be retagged in the diff below.
352         //Unfortunately, this doesn't produce the correspondence for the edges
353         labelEdges(samemap.lst[i].f, "inboth");
354 
355         var nf = samemap.lst[i].f,
356             nt = samemap.lst[i].t;
357         diffEdges(nf, nf.adjacencies, nt, nt.adjacencies);
358         diffEdges(nf, nf.incoming, nt, nt.incoming);
359 
360         //copy over any missing layout
361         if (nt.data.lentot) {
362             nf.data.lentot = nt.data.lentot;
363         }
364         if (nt.data.critpt) {
365             nf.data.critpt = nt.data.critpt;
366         }
367     }
368     
369     var addadj = [];
370     
371     //tag differences
372     for (var i = 0; i < diff.length; ++i) {
373         var d = diff[i];
374         if (d.t == null) {
375             //easy case: it's only in the from graph; edges are fine.
376             d.f.data.diff = "infrom";
377             //incoming/outgoing edges are only in from
378             labelEdges(d.f, "infrom");
379         } else {
380             var t = d.t;
381             t.data.diff = "into";
382             //copy over the node
383             grf.parsed.push(t);
384             grf.index[t.id] = t;
385             //attempt to map edges
386             var newadj = [];
387             for (var j = 0; j < t.adjacencies.length; ++j) {
388                 var adj = t.adjacencies[j];
389                 var copy = GraphUtil.dupedge(adj);
390                 var childid = (adj.nodeFrom == t.id) ? adj.nodeTo : adj.nodeFrom;
391                 if (childid in samemap.rev) {
392                     //we can map back to the from graph
393                     if (adj.nodeFrom == t.id) {
394                         copy.nodeTo = samemap.rev[childid].id;
395                     } else {
396                         copy.nodeFrom = samemap.rev[childid].id;
397                     }
398                 } //else keep the original edge
399                 copy.data.diff = "into";
400                 newadj.push(copy);
401             }
402             t.adjacencies = newadj;
403             
404             //look for missing incoming edges
405             for (var j = 0; j < t.incoming.length; ++j) {
406                 var adj = t.incoming[j];
407                 var childid = (adj.nodeFrom == t.id) ? adj.nodeTo : adj.nodeFrom;
408                 if (childid in samemap.rev) {
409                     var dupedge = {nodeFrom: samemap.rev[childid].id, nodeTo: t.id, data: adj.data};
410                     dupedge.data.diff = "into";
411                     addadj.push(dupedge);
412                 }
413             }
414             
415             if (d.f != null) {
416                 d.f.data.diff = "infrom";
417                 labelEdges(d.f, "infrom");
418             }
419         }
420     }
421     
422     //add missing incoming edges
423     for (var i = 0; i < addadj.length; ++i) {
424         var adj = addadj[i];
425         grf.index[adj.nodeFrom].adjacencies.push(adj);
426     }
427 
428     for (var i = 0; i < grf.parsed.length; ++i) {
429         grf.parsed[i].data.adjacencies = grf.parsed[i].adjacencies;
430         grf.parsed[i].data.incoming = grf.parsed[i].incoming;
431     }
432 
433     return grf;
434 }

436 /**
437  * @param graph the graph to compute
438  * @param options override vars for graph comparison
439  *      <dl>
440  *      <dt>cols</dt><dd>the existing map of node type -> color
441  *          (default {})</dd>
442  *      <dt>coldx</dt><dd>the index in the colormap to resume at
443  *          (default 0)</dd>
444  *      <dt>lblregex</dt><dd>the regex used to compute numbers from edges
445  *          (default match first floating point number)<dd>
446  *      </dl>
447  * @return a dict with:
448  *      <dl>
449  *      <dt>parsed</dt><dd>the parsed graph</dd>
450  *      <dt>index</dt><dd>map from node id to node</dd>
451  *      <dt>cols</dt><dd>the resulting colormap</dd>
452  *      <dt>coldx</dt><dd>color index we left off at</dd>
453  *      </dl>
454  */
455 RFGraph.makegraph = function(graph, options) {
456     var gr = graph;
457     var cols = getdefault(options, "cols", {});
458     var coldx = getdefault(options, "coldx", 0);
459 	var lblregex = getdefault(options, "lblregex", /\d+\.\d+/ig);
460     var edgetimedx = getdefault(options, "edgetimedx", 0); //index to parse the edge time from
461     var edgetimekey = getdefault(options, "edgetimekey", "weight"); //key to place the edge time in
462     var colorkey = getdefault(options, "colorkey", "color"); //key to place the edge color in
463 
464     //input bounding box
465     var bbox = [0,0,100,100];
466     var S_HEIGHT = 1000;
467     var S_WIDTH = 1300;
468     
469 	var index = {};
470 	var parsed = [];
471     
472     var topnode = null;
473     var topnodey = 1000;
474     
475 	for (var i = 0; i < gr.nodes.length; ++i) {
476         var tn = gr.nodes[i];
477         var initx = 0;
478         var inity = 0;
479         var pos = tn["pos"];
480         if (pos) {
481             var sdx = pos.indexOf(",");
482             initx = parseInt(pos.substr(0, sdx));
483             inity = parseInt(pos.substr(sdx+1));
484             initx = (initx - bbox[0]) * S_WIDTH / bbox[2] - S_WIDTH/2;
485             inity = S_HEIGHT/2 - (inity - bbox[1]) * S_HEIGHT / bbox[3];
486         }
487         
488         var lbl = tn["label"];
489         var fulllbl = lbl;
490         var lblcol = 0;
491         var eid = 0;
492         var tid = 0;
493         if (lbl) {
494             lbl = lbl.trim();
495             lblsp = lbl.split("__");
496             //length to start trimming lbl
497             var substlen = 0;
498             //match the eXXX or SSXXX
499             if (lblsp.length > 0) {
500                 var ematch = (/[A-Za-z]*(\d+)_?/ig).exec(lblsp[0]);
501                 //we found a numeric match
502                 if (ematch && ematch.length > 1) {
503                     eid = parseInt(ematch[1],10);
504                     if (eid in cols) {
505                         lblcol = cols[eid];
506                     } else {
507                         Util.info("Color not found for key ", eid);
508                         lblcol = cols[eid] = Const.COLORS[++coldx];
509                     }
510                     //if there was a __, add 2 to the length of the matched string
511                     substlen += ematch[0].length + ((lblsp.length > 1) ? 2 : 0);
512                 }
513                 ematch = (/[A-Za-z]*(\d+)/ig).exec(lblsp[1]);
514                 if (ematch && ematch.length > 1) {
515                     tid = parseInt(ematch[1],10);
516                     substlen += lblsp[1].length + 2;
517                 }
518 
519                 lbl = getReadableName(lbl, eid, tid, lbl.substring(substlen));
520             }
521         }
522         
523 		var thenew = {
524 			"id":tn["id"],
525 			"key":fulllbl,
526             "name":lbl,
527 			"adjacencies":[],
528             "incoming":[],
529             data: {
530                 "startx":initx,
531                 "starty":inity, //parseFloat(gr.nodes[i]["id"])
532                 "lblcol":lblcol,
533                 "nodeid":eid,
534                 "threadid":tid
535             }
536 		};
537         
538         //dot output includes a node with id="node" and one with id="graph"
539         if (tn["id"] == "graph") {
540             if (tn["bb"]) {
541                 bbox = tn["bb"].split(",").map(function(x) {return parseInt(x);});
542             }
543         } else if (tn["id"] == "node") {
544         
545         } else if (tn["id"] == "1") {
546             //cluster summary node
547         } else {
548             parsed.push(thenew);
549             index[thenew.id] = thenew;
550             
551             //find the "top" node if it was topmost in the layout
552             if (pos && inity < topnodey) {
553                 topnode = thenew;
554                 topnodey = inity;
555             }
556         }
557 	}
558     
559 	//pull out edges
560 	for (var i = 0; i < gr.edges.length; ++i) {
561         var te = gr.edges[i];
562 		var from = te.from;
563 		var to = te.to;
564         
565         //attempt to parse a microsecond number from the label
566         var edgetime = 10;
567         var edgetimes = [];
568         if (te.label) {
569             var lblm = te.label.match(lblregex);
570             //try the fallback
571             if (!lblm) {
572                 lblm = te.label.match(/([\d\.]+)/ig);
573             }
574             if (lblm) {
575                 for (var j = 0; j < lblm.length; ++j) {
576                     edgetimes.push(parseFloat(lblm[j]));
577                 }
578                 if (!edgetimes[edgetimedx]) {
579                     edgetime = edgetimes[0];
580                 } else {
581                     edgetime = edgetimes[edgetimedx];
582                 }
583                 edgetime = Math.max(0, edgetime);
584             } else {
585                 Util.log("Couldn't parse label", te.label);
586             }
587         }
588         
589         var ne = {
590             "nodeFrom": from,
591             "nodeTo": to,
592             "data": {
593                 "allweights": edgetimes
594             }
595         };
596         if (te.color) {
597             ne.data[colorkey] = te.color;
598         }
599         ne.data[edgetimekey] = edgetime;
600 
601         index[from].adjacencies.push(ne);
602         index[to].incoming.push(ne);
603 	}
604 	
605     //we couldn't get the top node from layout, get it through zero-length incoming
606     if (null == topnode) {
607         zeroins = [];
608         for (var i = 0; i < parsed.length; ++i) {
609             if (parsed[i].incoming.length == 0) {
610                 zeroins.push(parsed[i]);
611             }
612         }
613         if (zeroins.length > 0) {
614             info("Warning: more than one node with indegree 0!", zeroins);
615         }
616         topnode = zeroins[0];
617     }
618     
619     return {
620         "parsed": parsed,
621         "topnode": topnode,
622         "index": index,
623         "cols": cols,
624         "coldx": coldx
625     };
626 }

628 /**
629  * Get a "human-readable" version of a callpoint label
630  * @param original the original label
631  * @param eid the machine identifier
632  * @param tid the "t" identifier (?)
633  * @param lastpart the remaining string
634  * @return a string that will be displayed in the visualization
635  */
636 RFGraph.getReadableName = function(/**string*/ original, /**int*/ eid, /**int*/ tid, /**string*/ lastpart) {
637     //original: something like 'e10__t3__RPC_CALL_TYPE'
638     //lastpart: something like 'RPC_CALL_TYPE'
639 
640     //This is what we did before to make "readable" names.
641     //lastpart = lastpart.replace("_TYPE\n"," ");
642     //lastpart = lastpart.replace("DEFAULT","");
643 
644     var hname = lastpart;
