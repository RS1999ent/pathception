 31 /**
 32  * @param contsel selector for the scrollbar container
 33  * @param btnsel selector for the scrollbar handle
 34  * @param scrollfn lambda(val,delta)
 35  */
 36 var Scrollbar = function(contsel, btnsel, scrollfn) {
 37     this.contsel = contsel;
 38     this.btnsel = btnsel;
 39     this.setMax(1000);
 40     this.init();
 41     this.lastscroll = 0;
 42     this.scrollfn = scrollfn;
 43     this.viewable = this.getHeight();
 44 }
 45 Scrollbar.prototype.init = function() {
 46     var btn = $(this.btnsel);
 47     var self = this;
 48     var dehighlight = function() {
 49         $(self.btnsel).css({
 50             "background-color": "#666"
 51         });
 52     };
 53     var endfn = function(e) {
 54         $(document).unbind(".sbar");
 55         self.holding = false;
 56         if (!self.mouseIsOver) {
 57             dehighlight();
 58         }
 59     };
 60     var mousepos = function(e) {
 61         return e.pageY || (e.clientY + (document.documentElement.scrollTop || document.body.scrollTop));
 62     };
 63     btn.mousedown(function(e) {
 64         e.preventDefault();
 65         e.stopPropagation();
 66         var clickstart = e.offsetY;
 67         self.holding = true;
 68         $(document).bind("mousemove.sbar", function(e) {
 69             e.preventDefault();
 70             var height = self.getHeight();
 71             var newval = ((mousepos(e)-clickstart) / height) * self.max + self.getViewable() / 2;
 72             val = Math.max(0, Math.min(self.max, newval));
 73             self.doScroll(val);
 74         });
 75         $(document).bind("mouseup.sbar", function(e) {
 76             endfn(e);
 77         });
 78     });
 79     btn.mouseover(function(e) {
 80         $(self.btnsel).css({
 81             "background-color": "#000"
 82         });
 83         self.mouseIsOver = true;
 84     });
 85     btn.mouseout(function(e) {
 86         if (!self.holding) {
 87             dehighlight();
 88         }
 89         self.mouseIsOver = false;
 90     });
 91 
 92     $(document).bind("keydown.sbar", function(e) {
 93         if ((e.keyCode >= 37) && (e.keyCode <= 40)) {
 94             var delta = self.getViewable() / 10;
 95             if (e.keyCode == 38) { //up
 96                 self.doScroll(self.lastscroll - delta);
 97             } else if (e.keyCode == 40) { //down
 98                 self.doScroll(self.lastscroll + delta);
 99             }            
100         }
101     });
102 
103     var cont = $(this.contsel);
104 }
105 Scrollbar.prototype.updateHandlePos = function(val) {
106     var pos = (val - this.getViewable() / 2) / this.max * this.getHeight();
107     $(this.btnsel).css({
108         "top": pos
109     });
110     this.setMax(this.max);
111 }
112 Scrollbar.prototype.getHeight = function() {
113     return $(this.contsel).height();
114 }
115 Scrollbar.prototype.getViewable = function() {
116     return this.viewable;
117 }
118 Scrollbar.prototype.setViewable = function(val) {
119     this.viewable = val;
120     this.updateHandlePos(this.lastscroll);
121 }
122 Scrollbar.prototype.setMax = function(max) {
123     this.max = max;
124     var fracshow = this.getViewable() / max;
125     this.btnheight = this.getHeight() * (this.getViewable() / this.max);
126     $(this.btnsel).css("height", Math.min(this.getHeight(), this.btnheight));
127 }
128 Scrollbar.prototype.getMax = function() {
129     return this.max;
130 }
131 Scrollbar.prototype.doScroll = function(val) {
132     if (val != this.lastscroll) {
133         this.scrollfn(val, val - this.lastscroll);
134         this.lastscroll = val;
135         this.updateHandlePos(val);
136     }
137 }
138 Scrollbar.prototype.setVal = function(val) {
139     this.lastscroll = val;
140     this.updateHandlePos(this.lastscroll);
141 }
