 36 /**
 37 Distance from p1 to p2
 38 @param p1 a point with .x and .y
 39 @param p2 a point with .x and .y
 40 @return floating point euclidean distance
 41 */
 42 GraphUtil.dist = function(p1,p2) {
 43     return Math.sqrt((p2.x-p1.x)*(p2.x-p1.x)+(p2.y-p1.y)*(p2.y-p1.y));
 44 }

 46 /**
 47 Are two undirected edges equal? (based on nodeFrom and nodeTo)
 48 */
 49 GraphUtil.edgeobjeq = function(/**Edge*/ a,/**Edge*/ b) {
 50     if (a.nodeFrom.id == b.nodeFrom.id && a.nodeTo.id == b.nodeTo.id) {
 51         return true;
 52     }
 53     if (a.nodeFrom.id == b.nodeTo.id && a.nodeTo.id == b.nodeFrom.id) {
 54         return true;
 55     }
 56     return false;
 57 }

 59 /**
 60  * is n1 == n2? (based on name)
 61  */
 62 GraphUtil.noddeq = function(/**Node*/ n1, /**Node*/ n2) {
 63     return n1.name == n2.name;
 64 }
 65 /** 
 66  * is efrom == eto?

 69 GraphUtil.edgeeq = function(/**Edge*/ e1,e1dx,/**Edge*/ e2,e2dx) {

 73 /**
 74  * is n in arr? (based on name)
 75  * if yes, return the index
 76  * if no, return -1
 77  */
 78 GraphUtil.nodeinarr = function(/**Node*/ n,/**Node[]*/ arr) {
 79     for (var i = 0; i < arr.length; ++i) {
 80         if (noddeq(n, arr[i])) {
 81             return i;
 82         }
 83     }
 84     return -1;
 85 }
 86 /**
 87  * is n in arr? (based on id)
 88  * if yes, return the index
 89  * if no, return -1
 90  */
 91 GraphUtil.nodeidinarr = function(/**Node*/ n,/**Node*/ arr) {
 92     for (var i = 0; i < arr.length; ++i) {
 93         if (n.id == arr[i].id) {
 94             return i;
 95         }
 96     }
 97     return -1;
 98 }

100 /**
101  * is e in arr?
102  */
103 GraphUtil.edgeinarr = function(/**Edge*/ e, /**int*/ edx,/**Edge[]*/ arr,/**int*/ arrdx) {
104     for (var i = 0; i < arr.length; ++i) {
105         if (edgeeq(e,edx,arr[i],arrdx)) {
106             return i;
107         }
108     }
109     return -1;
110 }

112 /**
113  * get adjacent nodeset to n
114  * @param index string => Node map
115  */
116 GraphUtil.neighbors = function(/**Node*/ n,index) {
117     var nbr = [];
118     for (var i = 0; i < n.adjacencies.length; ++i) {
119         var adj = n.adjacencies[i];
120         nbr.push(index[(adj.nodeFrom == n.id) ? adj.nodeTo : adj.nodeFrom]);
121     }
122     return nbr;
123 }

148 /**
149  * Apply a breadth-first numbering to the given graph
150  * @param gr the graph to bfs-number (gr.topnode is taken to be the root)
151  */
152 GraphUtil.bfsnumber = function(gr) {
153     var stk = [gr.topnode];
154     var num = 0;
155     while (stk.length > 0) {
156         ++num;
157         var tp = stk.splice(0,1)[0];
158         var nbr = neighbors(tp, gr.index);
159         tp.data.bfsn = num;
160         
161         for (var i = 0; i < nbr.length; ++i) {
162             var n = nbr[i];
163             if (!n.data.inlinks) {
164                 n.data.inlinks = 0;
165             }
166             n.data.inlinks += 1;
167             if (!n.data.bfsn || n.data.bfsn == 0) {
168                 stk.push(n);
169             }
170         }
171     }
172 }

174 /**
175  * build a name -> node index for gr
176  * @param gr a graph
177  * @return map of string => int
178  */
179 GraphUtil.name2nodedx = function(gr) {
180     var ndx = {};
181     for (var i in gr.index) {
182         var name = gr.index[i].name;
183         if (!(name in ndx)) {
184             ndx[name] = [];
185         }
186         ndx[name].unshift(gr.index[i]);
187     }
188     return ndx;
189 }

191 var SIMPLEY_YSCALE = 0.02*10;
192 
193 /**
194 Basic recursive y position proportional to edge weight
195 @param index the node name to object map
196 @param root the node we're currently considering
197 */
198 GraphUtil.simpleypos = function(/**map*/ index, /**Node*/ root) {
199     for (var i = 0; i < root.adjacencies.length; ++i) {
200         var adj = root.adjacencies[i];
201         var len = 20;
202         if (adj.data.weight) {
203             len += adj.data.weight * SIMPLEY_YSCALE;
204         }
205         
206         var subn = index[root.id == adj.nodeFrom ? adj.nodeTo : adj.nodeFrom];
207         if (subn.data.startyset) {
208             subn.data.starty = Math.max(root.data.starty + len, subn.data.starty);
209         } else {
210             subn.data.starty = root.data.starty + len;
211             subn.data.startyset = true;
212         }
213         simpleypos(index, subn);
214     }
215 }

217 /**
218  * Apply y positions to nodes n[] in g based upon value in n.data["key"]
219  */
220 GraphUtil.applyY = function(/**Graph*/ g, /**string*/ key) {
221     for (var k in g.index) {
222         var n = g.index[k];
223         n.data.starty = n.data[key];
224     }
225 }
226 
227 /**
228  * Apply weighted y positions to nodes n[] in g based upon values 
229  *  in n.data[keys[0].key]...n.data[keys[k].key], weighted by keys[0].w...keys[k].w
230  */
231 GraphUtil.applyMultiY = function(/**Graph*/ g, /**Array[{key:string,w:Number}]*/ keys) {
232     for (var k in g.index) {
233         var n = g.index[k];
234         var yp = 0;
235         var wtotal = 0;
236         for (var i = 0; i < keys.length; ++i) {
237             if (n.data[keys[i].key] !== undefined) {
238                 yp += keys[i].w * n.data[keys[i].key];
239                 wtotal += keys[i].w;
240             }
241         }
242         yp /= wtotal;
243         if (n.data.diff != "g2") {
244             n.data.starty = yp;
245         }
246     }
247 }
