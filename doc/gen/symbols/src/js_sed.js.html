 36 /**
 37  * @param index node name to object map as created by {@link RFGraph.makegraph}
 38  * @param root the node we're currently considering
 39 */
 40 Sed.sortChildren = function(/**map*/ index, /**Node*/ root) {
 41     if (!root.data.sorted) {
 42         root.adjacencies = root.adjacencies.sort(function(a, b) {
 43             if (a.nodeTo === b.nodeTo) {
 44                 return 0;
 45             } else {
 46                 return (a.nodeTo < b.nodeTo) ? -1 : 1;
 47             }
 48         });
 49         root.data.sorted = true;
 50     }
 51     
 52     for (var i = 0; i < root.adjacencies.length; ++i) {
 53         var adj = root.adjacencies[i];
 54         var subn = index[root.id == adj.nodeFrom ? adj.nodeTo : adj.nodeFrom];
 55         if (!subn.data.sorted) {
 56             Sed.sortChildren(index, subn);
 57         }
 58     }
 59 }

 61 /**
 62  * Note: Adds traversal marks to the graph
 63  * @param options flattening params
 64  *      <dl>
 65  *      <dt>whole</dt><dd>the entire serialized list to append to (kept as a ref)</dd>
 66  *      <dt>dag</dt><dd>(set by dfsFlatten) if true, the graph is a dag, if false, it *might* not be.</dd>
 67  *      </dl>
 68  * @param index node name to object map
 69  * @param root the node we're currently considering
 70  */
 71 Sed.dfsFlatten = function(/**dict*/ options, /**map*/ index, /**Node*/ root) {
 72     if (!root.data.flattened) {
 73         options.whole.push(root);
 74         root.data.flattened = true;
 75     }
 76     
 77     //TODO: flipped DFS ordering until we do chain-insertion sensitive cost
 78     //  i.e. cost for subsequent insertions is lower than cost for point insertion
 79     //for (var i = 0; i < root.adjacencies.length; ++i) {
 80     for (var i = root.adjacencies.length - 1; i >= 0; --i) {
 81         var adj = root.adjacencies[i];
 82         var subn = index[root.id == adj.nodeFrom ? adj.nodeTo : adj.nodeFrom];
 83         if (root.id != adj.nodeFrom) {
 84             Util.debug("Warning: edge directionality incorrect from", root.id, adj.nodeFrom, adj.nodeTo);
 85         }
 86         if (!subn.data.flattened) {
 87             Sed.dfsFlatten(options, index, subn);
 88         } else if (subn.data.dfsdone) {
 89             options.dag = false;
 90         }
 91     }
 92     root.data.dfsdone = true; 
 93     
 94     return options;
 95 }

 97 /**
 98  * Both graphs must have been flattened to string.
 99  * same is a linked list of
100  *  f: node in source graph g1
101  *  t: node in destination graph g2
102  *  p: next pointer
103  * diff is a list of
104  *  f: node in source graph g1
105  *  t: node in destination graph g2
106  *  p1: position in source graph g1
107  *  p2: position in destination graph g2
108  * @param g1 graph "from"
109  * @param g2 graph "to"
110  * @return {
111  *  same: a linked list of nodes that are paired in both graphs
112  *  diff: a list of edits
113  * }
114  */
115 Sed.editDistance = function(g1,g2) {
116     var s1 = g1.flat;
117     var s2 = g2.flat;
118     /*
119     var min3t = function(a,b,c) {
120         if (a.length < b.length && a.length < c.length) {
121             return a;
122         } else if (b.length < c.length) {
123             return b;
124         } else {
125             return c;
126         }
127     };
128     */
129     
130     //allocate
131     var d = [];
132     for (var i = 0; i <= s1.length; ++i) {
133         d[i] = [];
134         for (var j = 0; j <= s2.length; ++j) {
135             d[i].push([]);
136         }
137     }
138     
139     /*
140     diff is a tuple of:
141         p: parent
142         l: length
143         v: value
144     same is a tuple of:
145         p: parent
146         f: from char
147         t: to char
148     */
149     var mkcell = function() {
150         return {same:null, diff:{p:null, l: 0, v:null}};
151     }
152     
153     var dst = mkcell();
154     for (var i = 0; i <= s1.length; ++i) {
155         d[i][0] = dst;
156         dst = {
157             same: dst.same,
158             diff: {p: dst.diff, l: dst.diff.l+1, v: {f: s1[i], t: null, p1: i, p2: 0}}
159         };
160     }
161     dst = mkcell();
162     for (var i = 0; i <= s2.length; ++i) {
163         d[0][i] = dst;
164         dst = {
165             same: dst.same,
166             diff: {p: dst.diff, l: dst.diff.l+1, v: {f: null, t: s2[i], p1: 0, p2: i}}
167         };
168     }
169     
170     for (var j = 1; j <= s2.length; ++j) {
171         for (var i = 1; i <= s1.length; ++i) {
172             var c1 = s1[i-1];
173             var c2 = s2[j-1];
174             if (c1.key === c2.key) {
175                 d[i][j] = {same: {f: c1, t: c2, p: d[i-1][j-1].same}, diff: d[i-1][j-1].diff};
176             } else {
177                 var m1 = d[i-1][j], d1 = {p: m1.diff, l: m1.diff.l+1, v:{f: c1, t: null, p1: i-1, p2: j}},
178                     m2 = d[i][j-1], d2 = {p: m2.diff, l: m2.diff.l+1, v:{f: null, t: c2, p1: i, p2: j-1}},
179                     m3 = d[i-1][j-1], d3 = {p: m3.diff, l: m3.diff.l+1, v:{f: c1, t: c2, p1: i-1, p2: j-1}};
180                 var m = null;
181                 
182                 if (d1.l < d2.l && d1.l < d3.l) {
183                     d[i][j] = {same: m1.same, diff: d1};
184                 } else if (d2.l < d3.l) {
185                     d[i][j] = {same: m2.same, diff: d2};
186                 } else {
187                     d[i][j] = {same: m3.same, diff: d3};
188                 }
189             }
190         }
191     }
192     
193     var rv = d[s1.length][s2.length];
194     rv.diff = Sed.unrollDiff(rv.diff);
195     return rv;
196 }

198 /**
199  * unroll a "same" linked list into a bidimap of ids in each graph to nodes in the other.
200  */
201 Sed.unrollSame = function(same) {
202     var lst = [];
203     var fdmap = {};
204     var revmap = {};
205     var n = same;
206     while (null != n) {
207         lst.push({f: n.f, t: n.t});
208         fdmap[n.f.id] = n.t;
209         revmap[n.t.id] = n.f;
210         n = n.p;
211     }
212     return {
213         lst: lst,
214         fd: fdmap,
215         rev: revmap
216     };
217 }

219 /**
220  * unroll a "diff" linked list into a list of f/t/p1/p2 pairs
221  */
222 Sed.unrollDiff = function(diff) {
223     var lst = [];
224     var d = diff;
225     while (null != d.p) {
226         lst.unshift(d.v);
227         d = d.p;
228     }
229     return lst;
230 }
