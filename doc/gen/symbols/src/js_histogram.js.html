 36 /**
 37  * Compute a histogram of edge lengths for the given graph
 38  * @param nodes the list of graph nodes
 39  * @param bw the bin width
 40  */
 41 Histogram.disthistRaw = function(/**Node[]*/ nodes, /**int*/ bw) {
 42     var hist = [];
 43     for (var i = 0; i < nodes.length; ++i) {
 44         for (var j = 0; j < nodes[i].adjacencies.length; ++j) {
 45             var adj = nodes[i].adjacencies[j];
 46             if (adj.data.weight) {
 47                 var bdx = Math.floor(adj.data.weight / bw);
 48                 if (hist[bdx]) {
 49                     hist[bdx] += 1;
 50                 } else {
 51                     hist[bdx] = 1;
 52                 }
 53             }
 54         }
 55     }
 56     return hist;
 57 }
 58 
 59 Histogram.histtopx = function(/**int[]*/ hist) {
 60     var px = 0;
 61     var map = [];
 62     for (var i = 0; i < hist.length; ++i) {
 63         if (hist[i]) {
 64             for (var j = px; j < px + hist[i]; ++j) {
 65                 map[j] = i;
 66             }
 67             px += hist[i];
 68         }
 69     }
 70     
 71     return map;
 72 }

 74 Histogram.LOG = function(dta) { return Math.log(dta)/Math.log(10); }
 75 Histogram.DELOG = function(dta) { return Math.pow(10,dta); }
 76 Histogram.KF_F = function(dta) { return dta.weightf; }
 77 Histogram.KF_LOGF = function(dta) { return Histogram.LOG(Histogram.KF_F(dta)+1); }
 78 Histogram.KF_T = function(dta) { return dta.weightt; }
 79 Histogram.KF_LOGT = function(dta) { return Histogram.LOG(Histogram.KF_T(dta)+1); }
 80 Histogram.KF_D = function(dta) { return dta.weightt - dta.weightf; }
 81 Histogram.KF_LOGD = function(dta) {
 82     var d = Histogram.KF_D(dta);
 83     if (isNaN(d)) {
 84         return NaN;
 85     } else {
 86         return Histogram.LOG(Math.abs(d) + 1) * ((d < 0) ? -1 : 1);
 87     }
 88 }

 90 /**
 91  * Find maximum/minimum length for precursor/mutation/individual-edge-difference
 92  *  edges in the graph
 93  */
 94 Histogram.findBounds = function(
 95     /**Graph*/ g,
 96     /**Function*/ keyfn) {
 97     var bounds = {min:1e20,max:0};
 98 
 99     for (var i = 0; i < g.parsed.length; ++i) {
100         for (var j = 0; j < g.parsed[i].adjacencies.length; ++j) {
101             var adjd = g.parsed[i].adjacencies[j].data;
102             var val = keyfn(adjd);
103             if (!isNaN(val)) {
104                 bounds.max = Math.max(bounds.max, val);
105                 bounds.min = Math.min(bounds.min, val);
106             }
107         }
108     }
109 
110     return bounds;
111 }

113 /**
114  * @param g graph to compute histogram over
115  * @param keyfn function to extract a key
116  * @param b bounds with {max:,min:}
117  * @param nbins number of bins for histogram
118  */
119 Histogram.compute = function(
120     /**Graph*/ g,
121     /**Function*/ keyfn,  
122     /**Bounds*/ b, 
123     /**int*/ nbins) {
124     var bins = [];
125     var bdiff = b.max - b.min;
126     for (var i = 0; i < nbins + 1; ++i) {
127         bins.push([i/nbins*bdiff+b.min, 0]);
128     }
129 
130     for (var i = 0; i < g.parsed.length; ++i) {
131         for (var j = 0; j < g.parsed[i].adjacencies.length; ++j) {
132             var adjd = g.parsed[i].adjacencies[j].data;
133             var val = keyfn(adjd);
134             var bindx = Math.floor((val-b.min)/bdiff*nbins);
135             if (!isNaN(val) && !isNaN(bindx)) {
136                 bins[bindx][1] += 1;
137             }
138         }
139     }
140 
141     return bins;
142 }

144 Histogram.cumsum = function(bins) {
145     var newbins = [];
146     var sum = 0;
147     for (var i = 0; i < bins.length; ++i) {
148         sum += bins[i][1];
149         newbins.push([bins[i][0],sum]);
150     }
151     return newbins;
152 }
153 /**
154  * Normalize a CDF
155  */
156 Histogram.normalize = function(bins) {
157     var newbins = [];
158     var sum = 0;
159     var maxval = bins[bins.length-1][1];
160     for (var i = 0; i < bins.length; ++i) {
161         newbins.push([bins[i][0],bins[i][1]/maxval]);
162     }
163     return newbins;   
164 }
